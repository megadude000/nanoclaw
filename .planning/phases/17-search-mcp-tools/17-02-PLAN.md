  <behavior>
    isVaultPath(query: string): boolean
    - Returns true if query ends with '.md' OR starts with 'Areas/', 'Calendar/', or 'System/'
    - Returns false for all other strings

    checkConfidenceFirewall(level: string, domain: string, qdrant: QdrantClient): Promise<boolean>
    - If level is 'L10': return false immediately (no scroll call — L10 has no parent)
    - For L20+: parentLevel = 'L' + (parseInt(level.slice(1)) - 10)
    - Call qdrant.scroll('cortex-entries', { filter: { must: [level filter, domain filter, confidence any(['medium','high'])] }, limit: 1, with_payload: false })
    - Return true if points.length === 0 (blocked), false if points.length > 0 (allowed)

    buildSearchHandler({ qdrant, openai, vaultRoot }): async (args) => McpResult
    - If isVaultPath(args.query): read file from vaultRoot + args.query, return content
    - Else: embed args.query via openai.embeddings.create(), build filter from args.project/cortex_level/domain, call qdrant.search(), return formatted results array as JSON string
    - Limit capped at min(args.limit ?? 5, 20)
    - Result format: JSON array of { path, score, level, domain, project }

    buildReadHandler({ vaultRoot }): async (args) => McpResult
    - Resolve path.resolve(vaultRoot, args.path)
    - If resolved does not startsWith(vaultRoot + '/'): return isError: true "path traversal not allowed"
    - If !fs.existsSync(resolved): return isError: true "Not found: {args.path}"
    - Else: return fs.readFileSync(resolved, 'utf-8') as text content

    buildWriteHandler({ qdrant, writeIpc }): async (args) => McpResult
    - Parse frontmatter with gray-matter
    - Validate with inlined CortexFieldsStrict Zod schema (4 fields: cortex_level, confidence, domain, scope)
    - If validation fails: return isError: true with field errors listed
    - If cortex_level is L20 or higher: call checkConfidenceFirewall(cortex_level, domain, qdrant)
    - If firewall blocks: return isError: true with message "Firewall: L{N-10} entries for domain '{domain}' lack medium+ confidence"
    - Else: call writeIpc({ type: 'cortex_write', path: args.path, content: args.content, timestamp: ... })
    - Return success "Entry queued for write: {args.path}"
  </behavior>
